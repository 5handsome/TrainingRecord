题目描述
输入n (n<5000000 且 n 为奇数) 个数字 （最大为10的9次方）和k  输出这些数字的第 k 小的数。最小的数是第 0 小。
请尽量不要使用 nth_element 来写本题，因为本题的重点在于练习分治算法

题目分析
	最开始我以为只是简单输入并排序 输出a[k+1] 但是tle了 所以用了快排
	主要代码void qs(int r1,int r2)//输入边界
	{
		int t1 = r1,t2 = r2;
		int mid = a[(r1+r2)/2];
		while(t1<=t2)//将中间的左边变成小于mid 右边大于mid
		{
			while(mid > a[t1]) t1 ++;
			while(mid < a[t2])t2 --;
			if(t1<=t2)
			{
				swap(a[t1],a[t2]);
				t1 ++;t2 --;//更新t1 t2 避免死循环 也方便下一步操作
			}
		}

		if(k <= t2) return qs(r1,t1);//最后会有 r1<=t2<t1<=r2 所以小于t2在左边界搜索
		else if(k >= t1) return qs(t2,r2);
		else cout << a[t2+1];
	}



#include<iostream>
using namespace std;
int a[5000005],k;

void qs(int r1,int r2)
{
	int t1 = r1,t2 = r2;
	int mid = a[(r1+r2)/2];
	while(t1<=t2)
	{
		while(mid > a[t1]) t1 ++;
		while(mid < a[t2])t2 --;
		if(t1<=t2)
		{
			swap(a[t1],a[t2]);
			t1 ++;t2 --;
		}
	}

	if(k <= t2) return qs(r1,t1);
	else if(k >= t1) return qs(t2,r2);
	else cout << a[t2+1];
}
int main()
{
	int n;
	cin >> n >> k;
	for(int i = 0;i < n;i ++)
		cin >> a[i];
	qs(0,n-1);
}
