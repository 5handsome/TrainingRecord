# 拼数
> [拼数](https://www.luogu.com.cn/problem/P1012)

````cpp
#include <algorithm>
#include <iostream>
#include <vector>
using namespace std;

template <typename it>
void mySort(it begin, it end) {
    sort(begin, end);

    while (begin != end - 1) {
        it next = begin + 1;
        if ((*next).length() > (*begin).length()) {
            int i = 0;
            while (i < begin->length() && (*next)[i] == (*begin)[i]) i++;
            if (i == begin->length()) {
                if ((*next)[i] < (*begin)[0]) {
                    string buf = *begin;
                    *begin = *next;
                    *next = buf;
                }
            }
        }
        begin++;
    }
}

int main() {
    int num;
    vector<string> strs;
    cin >> num;
    while (num--) {
        string str;
        cin >> str;
        strs.push_back(str);
    }
    mySort(strs.begin(), strs.end());
    for (int i = strs.size() - 1; i >= 0; i--) {
        cout << strs.at(i);
    }
    return 0;
}
````
