#【深基9.例4】求第 k 小的数
> [深基9.例4】求第 k 小的数](https://www.luogu.com.cn/problem/P1923)

````cpp
#include <cstdio>
using namespace std;
int main() {
    int cnt, order;
    int num;
    scanf("%d %d", &cnt, &order);

    auto sink = [](int* arr, int now, int length) {
        for (int up = now, down = now * 2 + 1; down < length; up = down, down = up * 2 + 1) {
            if (down + 1 < length && arr[down] < arr[down + 1]) down++;
            if (arr[down] > arr[up]) {
                int buf = arr[up];
                arr[up] = arr[down];
                arr[down] = buf;
            }
        }
    };

    order++;
    int* bigHeap = new int[order];
    for (int i = 0; i < order; i++) {
        scanf("%d", &num);
        bigHeap[i] = num;
    }
    for (int i = order / 2 - 1; i >= 0; i--)
        sink(bigHeap, i, order);

    while (cnt-- - order) {
        scanf("%d", &num);
        if (num < bigHeap[0]) {
            bigHeap[0] = num;
            sink(bigHeap, 0, order);
        }
    }

    printf("%d", bigHeap[0]);

    return 0;
}
````
