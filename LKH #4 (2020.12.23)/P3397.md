# 地毯

> [地毯](https://www.luogu.com.cn/problem/P3397) 

````cpp
#include <cstdio>
using namespace std;

int main() {
    int rc, cnt;
    scanf("%d %d", &rc, &cnt);
    int **arr = new int *[rc];
    for (int i = 0; i < rc; i++) {
        arr[i] = new int[rc];
        for (int j = 0; j < rc; j++) {
            arr[i][j] = 0;
        }
    }

    int a, b, x, y;
    while (cnt--) {
        scanf("%d %d %d %d", &a, &b, &x, &y);
        a--;
        b--;
        x--;
        y--;
        arr[a][b]++;
        if (x + 1 < rc)
            arr[x + 1][b]--;
        if (y + 1 < rc)
            arr[a][y + 1]--;
        if (x + 1 < rc && y + 1 < rc)
            arr[x + 1][y + 1]++;
    }

    for (int i = 0; i < rc; i++) {
        for (int j = 0; j < rc; j++) {
            if (i - 1 >= 0 && j - 1 >= 0)
                arr[i][j] -= arr[i - 1][j - 1];
            if (i - 1 >= 0)
                arr[i][j] += arr[i - 1][j];
            if (j - 1 >= 0)
                arr[i][j] += arr[i][j - 1];
            printf("%d ", arr[i][j]);
        }
        printf("\n");
    }

    return 0;
}
````
