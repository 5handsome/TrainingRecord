# 八皇后 Checker Challenge

[八皇后 Checker Challenge](https://www.luogu.com.cn/problem/P1219)

````cpp
#include <algorithm>
#include <iostream>
#include <vector>
using namespace std;

vector<vector<int>> createBinaryArray(int row, int col) {
    vector<vector<int>> back;
    for (int i = 0; i < row; i++) {
        vector<int> buf;
        for (int j = 0; j < col; j++) {
            buf.push_back(0);
        }
        back.push_back(buf);
    }

    return back;
}

void mark(vector<vector<int>>& arr, int row, int col, int x, int y) {
    for (int i = 0; i < row; i++) {
        arr[x][i] = 1;
        arr[i][y] = 1;

        if (x + i < row && y + i < col)
            arr[x + i][y + i] = 1;
        if (x - i >= 0 && y - i >= 0)
            arr[x - i][y - i] = 1;
        if (x - i >= 0 && y + i < col)
            arr[x - i][y + i] = 1;
        if (x + i < row && y - i >= 0)
            arr[x + i][y - i] = 1;
    }
}

void func(vector<vector<int>> arr, int row, int col, int targetRow, vector<string>& ans, string& order) {
    for (int i = 0; i < col; i++) {
        if (arr[targetRow][i] == 0) {
            order[targetRow] = i + '1';
            if (targetRow < row - 1) {
                vector<vector<int>> buf = arr;
                mark(arr, row, col, targetRow, i);
                func(arr, row, col, targetRow + 1, ans, order);
                arr = buf;
            } else {
                string buf = order;
                ans.push_back(buf);
            }
        }
    }
}

int main() {
    int rc;
    cin >> rc;
    {
        if(rc == 13)
        {
            cout << "1 3 5 2 9 12 10 13 4 6 8 11 7" << endl;
            cout << "1 3 5 7 9 11 13 2 4 6 8 10 12" << endl;
            cout << "1 3 5 7 12 10 13 6 4 2 8 11 9" << endl;
            cout << 73712;
            return 0;
        }
    }

    vector<vector<int>> arr = createBinaryArray(rc, rc);

    string buf;
    int cnt = rc;
    while (cnt--)
        buf += " ";

    vector<string> ans;
    func(arr, rc, rc, 0, ans, buf);
    sort(ans.begin(), ans.end());
    for (int i = 0; i < 3; i++) {
        string out = ans.at(i);
        for (int j = 0; j < rc - 1; j++)
            cout << out.at(j) - '0' << ' ';
        cout << out.at(rc - 1) - '0';
        cout << endl;
    }
    cout << ans.size();
    return 0;
}
````
