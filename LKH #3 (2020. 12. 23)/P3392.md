# 涂国旗

[涂国旗](https://www.luogu.com.cn/problem/P3392)

````cpp
#include <algorithm>
#include <iostream>
#include <vector>
using namespace std;
int main() {
    int row, col;
    cin >> row >> col;

    int* redSum = new int[row];
    int* blueSum = new int[row];
    int* whiteSum = new int[row];

    char c;
    char** flag = new char*[row];
    for (int i = 0; i < row; i++) {
        flag[i] = new char[col];
        redSum[i] = 0;
        blueSum[i] = 0;
        whiteSum[i] = 0;
        if (i > 0) {
            redSum[i] += redSum[i - 1];
            blueSum[i] += blueSum[i - 1];
            whiteSum[i] += whiteSum[i - 1];
        }
        for (int j = 0; j < col; j++) {
            cin >> c;
            flag[i][j] = c;
            if (c == 'W')
                whiteSum[i]++;
            else if (c == 'B')
                blueSum[i]++;
            else if (c == 'R')
                redSum[i]++;
        }
    }

    vector<int> sum;
    for (int i = 1; i < row; i++) {
        for (int j = i + 1; j < row; j++) {
            sum.push_back(whiteSum[i - 1] + blueSum[j - 1] - blueSum[i - 1] + redSum[row - 1] - redSum[j - 1]);
        } 
    }

    sort(sum.begin(), sum.end());
    cout << row * col - sum.at(sum.size() - 1);

    return 0;
}
````
