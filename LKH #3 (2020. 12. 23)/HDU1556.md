# Color the ball
 
 [Color the ball](http://acm.hdu.edu.cn/showproblem.php?pid=1556)
 
 ````cpp
 #include <iostream>
using namespace std;

int main() {
    int num;
    int a, b;
    while (cin >> num) {
        if (num == 0) break;
        int arr[num] = {
            0,
        };
        for (int i = 0; i < num; i++) {
            cin >> a >> b;
            arr[a - 1]++;
            arr[b]--;
        }
        for (int i = 0; i < num - 1; i++) {
            arr[i + 1] += arr[i];
            cout << arr[i] << ' ';
        }
        cout << arr[num - 1];
        cout << '\n';
    }
    return 0;
}
 ````
