# 数楼梯

[数楼梯](https://www.luogu.com.cn/problem/P1255)

````cpp
#include <iostream>
#include <vector>
using namespace std;

class BigNum {
    vector<int> num;

   public:
    BigNum(string s) {
        for (int i : s) {
            num.insert(num.begin(), i - '0');
        }
    };

    BigNum(const BigNum& rhs) {
        this->num = rhs.num;
    }
    BigNum operator+(const BigNum& rhs) {
        BigNum back(*this);
        int length;
        if (rhs.num.size() > back.num.size())
            length = rhs.num.size();
        else
            length = back.num.size();
        for (int i = 0; i < length; i++)
            back.setBit(i, back.getBit(i) + rhs.getBit(i));
        back.removeZero();
        return back;
    }

   private:
    void setBit(int index, int value) {
        while (index >= num.size())
            num.push_back(0);
        if (value > 9) {
            setBit(index + 1, value / 10 + getBit(index + 1));
            value %= 10;
        }
        num.at(index) = value;
    }

    int getBit(int index) const {
        if (index >= num.size())
            return 0;
        return num[index];
    }

    void removeZero() {
        while (num.at(num.size() - 1) == 0)
            num.erase(num.end() - 1);
    }

   public:
    string toString() {
        string back;
        for (int i : num)
            back.insert(back.begin(), i + '0');
        return back;
    }
};

int main() {
    vector<BigNum> arr;
    arr.push_back(BigNum("1"));
    arr.push_back(BigNum("1"));
    int num;
    cin >> num;
    if (num == 0) {
        cout << 0;
        return 0;
    }
    for (int i = 2; i <= num; i++)
        arr.push_back(arr.at(i - 1) + arr.at(i - 2));
    cout << arr.at(num).toString();
    return 0;
}
````
