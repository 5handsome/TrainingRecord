#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>
#include<iostream>
#include<vector>
#include<string>
#include<stdlib.h>
#include<algorithm>
using namespace std;

#define N 1010
static int n[1010][1010];

typedef struct location {
	int x;
	int y;
};

void Area(location a ,location b) {
	for (int i = a.y - 1; i < b.y; i++)
	{
		for (int j = a.x - 1; j < b.x; j++)
		{
			n[i][j]++;
		}
	}
}

int main() {
	int c, m;
	location m1, m2;
	cin >> c >> m;
	for (int i = 0; i < m; i++)
	{
		cin >> m1.x >> m1.y;
		cin >> m2.x >> m2.y;
		Area(m1, m2);
		
	}
	for (int i = 0; i < c; i++)
	{
		for (int j = 0; j < c; j++)
		{
			cout << n[j][i] << " ";
		}
		cout << endl;
	}
	return 0;
}
