#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>
#include<iostream>
#include<vector>
#include<string>
#include<stdlib.h>
#include<algorithm>
using namespace std;

bool cmp(string a, string b) {
	return a + b > b + a;
}

int main() {
	int n;
	string s[25];
	scanf("%d", &n);
	for (int i = 0; i < n; i++)
	{
		cin >> s[i];
	}
	sort(s, s + n, cmp);
	for (int i = 0; i < n; i++)
	{
		cout << s[i];
	}
	return 0;
}
