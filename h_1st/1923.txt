#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>
#include<iostream>
#include<vector>
#include<string>
#include<stdlib.h>
#include<algorithm>
using namespace std;

#define N 5000010

void Qsort(int a[], int l, int r) {
	if (l >= r) return;
	int i = l - 1, j = r + 1, mid = a[(l + r) / 2];
	while (i < j) {
		do i++; while(a[i] < mid);
		do j--; while (a[j] > mid);
		if (i < j)swap(a[i], a[j]);
	}
	Qsort(a, l, j);
	Qsort(a, j + 1, r);
}


int main() {
	int a[N], n, k;
	scanf("%d%d", &n, &k);
	for (int i = 0; i < n; i++)
	{
		scanf("%d", &a[i]);
	}
	Qsort(a, 0, n - 1);
	cout << a[k];

	return 0;
}
