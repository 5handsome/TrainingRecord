  正确代码是。
  #include<stdio.h>
#include<string.h>
#include<stdlib.h>
int main (void)
{
    int n,i;
    char a[101];
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        scanf("%s",a);
        if(a[0]==')'||a[strlen(a)-1]=='(')
            printf("NO\n");
        else if(strlen(a)%2==0)
            printf("YES\n");
            else
            printf("NO\n");
    }
    return 0;
}
